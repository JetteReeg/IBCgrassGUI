134 double CPlant::ReproGrow(double uptake) {
135  double SpacerRes, SeedRes, VegRes, dm_seeds, dummy1;
136  //fixed Proportion of resource to seed production
137  if (mRepro <= Traits->AllocSeed * mshoot)
138  {
139  // get resources allocated to seeds
140  SeedRes = uptake * Traits->AllocSeed;
141  // get resources allocated to spacers
142  SpacerRes = uptake * Traits->AllocSpacer;
143  // current week
144  int pweek = CEnvir::week;
145  //during the seed-production-weeks
146  if ((pweek >= Traits->FlowerWeek) &&
147  (pweek < Traits->DispWeek) &&
148  (((CTDPlant*)this)->GetAge() > 5)
149  ) {
150  // seed production
151  dm_seeds = dmGrow( Traits->growth * SeedRes,0);
152  mRepro += dm_seeds;
153 
154  // clonal growth
155  // for large AllocSeed, ressources may be < SpacerRes, then only take remaining ressources
156  dummy1 = max(0.0, min(SpacerRes, uptake - SeedRes));
157  mReproRamets += dmGrow( Traits->growth * dummy1,0);
158  // calculate resources left for vegetative growth
159  VegRes = uptake - SeedRes - dummy1;
160  } else {
161  // just do vegetative reproduction
162  VegRes = uptake - SpacerRes;
163  mReproRamets += dmGrow(Traits->growth * SpacerRes,0);
164  }// end else
165  } else // if there are already enough resources for seeds
166  VegRes = uptake;
167  // finally return resources for vegetative growth
168  return VegRes;
169 } //end reprogrow

174 void CPlant::SpacerGrow()
175 {
176  double mGrowSpacer=0;
177  int SpacerListSize=this->growingSpacerList.size();
178 
179  if (SpacerListSize==0)return;
180  // if there are resources for ramet growth
181  if ((mReproRamets>0))
182  {
183  // possible reduction by herbicide
184  mReproRamets*=AllocSpacerRed();
185  // resources for one spacer
186  mGrowSpacer=(mReproRamets/SpacerListSize);//resources for one spacer
187  //loop for all growing Spacer of one plant
188  for (int g=0; g<(SpacerListSize); g++)
189  {
190  // link to spacer
191  CPlant* Spacer = this->growingSpacerList[g];
192  // grow spacer
193  double lengthtogrow=Spacer->SpacerlengthToGrow;
194  lengthtogrow-=(mGrowSpacer/Traits->mSpacer);
195  Spacer->SpacerlengthToGrow=max(0.0,lengthtogrow);
196 
197  // Establishment for all growing Spacers in the last week of the year
198  if ((CEnvir::week==CEnvir::WeeksPerYear)
199  && (Spacer->SpacerlengthToGrow>0))
200  {
201  // get cell to establish in
202  double direction=Spacer->Spacerdirection;
203  double complDist=Spacer->Spacerlength;//should be positive
204  double dist=(complDist-Spacer->SpacerlengthToGrow);
205  double CmToCell=1.0/SRunPara::RunPara.CellScale();
206  int x2=CEnvir::Round(this->cell->x+cos(direction)*dist*CmToCell);
207  int y2=CEnvir::Round(this->cell->y+sin(direction)*dist*CmToCell);
208  //boundary condition
209  Boundary(x2,y2);
210 
211  Spacer->xcoord=x2/CmToCell;
212  Spacer->ycoord=y2/CmToCell;
213  Spacer->SpacerlengthToGrow=0;
214  } //end if pweek==WeeksPerYear
215  } //end List of Spacers
216  }// end if resources
217  // reset resources
218  mReproRamets=0;
219 } //end SpacerGrow

244 void CPlant::Grow()
245 {
246  double dm_shoot, dm_root,alloc_shoot;
247  double LimRes, ShootRes, RootRes, VegRes, ShootRes_tmp2, ShootRes_tmp, RootRes_tmp, RootRes_tmp2;
249  /********************************************/
250  /* dm/dt = growth*(c*m^p - m^q / m_max^r) */
251  /********************************************/
252  // which resource is limiting growth ?
253  LimRes=min(Buptake,Auptake);
254  // only take the limiting resource
255  VegRes=ReproGrow(LimRes);
256  //allocation to shoot and root growth
257  alloc_shoot= Buptake/(Buptake+Auptake);
258  // resources for shoot growth
259  ShootRes=alloc_shoot*VegRes;
260  // resources for root growth
261  RootRes =VegRes-ShootRes;
262  // calculate new allocation according to PFT traits allocroot/allocshoot
263  ShootRes_tmp2=ShootRes;
264  RootRes_tmp2=RootRes;
265 
266  ShootRes_tmp = ShootRes*this->Traits->allocroot;
267  RootRes_tmp = RootRes*this->Traits->allocshoot;
268  // if lower shoot growth
269  if(ShootRes_tmp<ShootRes) {
270  RootRes_tmp2=RootRes+(ShootRes-ShootRes_tmp);
271  ShootRes_tmp2=ShootRes_tmp;
272  }
273  // if lower root growth
274  if(RootRes_tmp<RootRes) {
275  ShootRes_tmp2=ShootRes+(RootRes-RootRes_tmp);
276  RootRes_tmp2=RootRes_tmp;
277  }
278  // update resources
279  ShootRes = ShootRes_tmp2;
280  RootRes = RootRes_tmp2;
281  //Shoot growth
282  dm_shoot=this->ShootGrow(ShootRes);
283  //Root growth
284  dm_root=this->RootGrow(RootRes);
285  // update shoot mass
286  // including potential herbicide effects (acts only on gain!)
287  mshoot+=dm_shoot*AdBioRed();
288  // update root mass
289  mroot+=dm_root;
290  // check if plant is stressed
291  if (stressed())++stress;
292  else if (stress>0) --stress;
293 } // end grow()

300 double CPlant::ShootGrow(double shres){
301  double Assim_shoot, Resp_shoot;
302  // growth function exponents
303  double p=2.0/3.0, q=2.0, r=4.0/3.0;
304  // growth limited by maximal resource per area -> similar to uptake limitation
305  Assim_shoot=Traits->growth*min(shres,Traits->Gmax*Ash_disc);
306  // respiration proportional to mshoot^2
307  Resp_shoot=Traits->growth*Traits->SLA
308  *pow(Traits->LMR,p)*Traits->Gmax
309  *pow(mshoot,q)/pow(Traits->MaxMass,r);
310  // return mass gain
311  return max(0.0,Assim_shoot-Resp_shoot);
312 }

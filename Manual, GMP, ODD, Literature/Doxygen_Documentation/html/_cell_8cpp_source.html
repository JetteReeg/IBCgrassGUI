233  //if cell is the center of a plant return 1

235  string ltype=PlantInCell->pft();
236  if (ltype==type) return 1;}

243  // set number of alive plants to number of plants in cell
244  unsigned int nonodead = noplants; // nb of not dead plants
245  // set number of PFT type in cell to 0
246  unsigned int notype = 0; // nb of type-plants
247  // go through all plants covering the cell
248  for (unsigned int i=0;i<noplants;i++){

252  if (AbovePlantList.at(i)->pft()==type) ++notype;
253  }// end for all plants in cell
254  // relative amount in the cell
255  // notype is the number of plant individuals of the specific type covering the cell
256  // nonodead is the number of not dead plant individuals covering the cell (overall)
257  if (nonodead>0)return notype/(double)nonodead;
258  // if all plants are dead reaturn 0
259  else return 0;
260  }// end above plant list is not empty

303 void CCell::AboveComp()
304 {
305  // if there are no plants covering the cell
306  if (AbovePlantList.empty()) return;
307  // if there is only one plant covering the cell
308  if (AbovePlantList.size()==1) {
309  // plant takes up all resources
310  AbovePlantList.back()->Auptake+=AResConc;
311  return;
312  }// end if only one plant covering the cell
313  // if mode of competition is asymmetric
314  if (SRunPara::RunPara.AboveCompMode==asymtot){
315  // only for above ground competition
316  // sort plant list by growth form/shoot length
317  sort(AbovePlantList.begin(),AbovePlantList.end(),CPlant::CompareShoot);
318  // get the biggest plant
319  CPlant* plant=*AbovePlantList.begin();
320  // biggest plant get all resources
321  plant->Auptake+=AResConc;
322  return;
323  }// end if asymmetric aboveground
324 
325  int symm=1;
326  // if partial asymetric competition
327  if (SRunPara::RunPara.AboveCompMode==asympart) symm=2;
328  double comp_tot=0, comp_c=0;
329 
330  //1. sum of resource requirement
331  // go through all plants
332  for (plant_iter iter=AbovePlantList.begin(); iter!=AbovePlantList.end(); ++iter){
333  // link to a plant
334  CPlant* plant=*iter;
335  // aboveground competition for resources depending on the model version
336  comp_tot+=plant->comp_coef(1,symm)
337  *prop_res(plant->pft(),1,SRunPara::RunPara.Version);
338  }// end over all plants
339 
340  //2. distribute resources
341  // go through all plants
342  for (plant_iter iter=AbovePlantList.begin(); iter!=AbovePlantList.end(); ++iter){
343  // link to plant
344  CPlant* plant=*iter;
345  // get specific competition
346  comp_c=plant->comp_coef(1,symm)
347  *prop_res(plant->pft(),1,SRunPara::RunPara.Version);
348  // set resource uptake
349  plant->Auptake+=AResConc*comp_c/comp_tot;
350  }// end over all plants

407 double CCell::prop_res(const string type,const int layer,const int version)const{
408  switch (version){
409  //no difference between intra- and interspecific competition
410  case 0: return 1;
411  //higher effects of intraspecific competition
412  case 1: switch (SRunPara::RunPara.ModelVersion){
413  // intensity of densitiy dependence in background mortality
414  // case 1: increasing intraspecific competition and density dependent mortality
415  // case 2: only density dependent mortality
416  // case3: no effect; same as Wei et al (2014)
417  case 1: if (layer==1){
418  map<string,int>::const_iterator noa =PftNIndA.find(type);
419  if (noa!=PftNIndA.end())
420  return 1.0/pow(noa->second,1);
421  }
422  // belowground layer
423  if (layer==2){
424  map<string,int>::const_iterator nob =PftNIndB.find(type);
425  if (nob!=PftNIndB.end())
426  return 1.0/pow(nob->second,1);
427  }break;
428  //
429  case 2: if (layer==1){
430  map<string,int>::const_iterator noa =PftNIndA.find(type);
431  if (noa!=PftNIndA.end())
432  return 1.0/pow(noa->second,0.5);
433  }
434  // belowground layer
435  if (layer==2){
436  map<string,int>::const_iterator nob =PftNIndB.find(type);
437  if (nob!=PftNIndB.end())
438  return 1.0/pow(nob->second,0.5);
439  }break;
440  case 3: if (layer==1){
441  map<string,int>::const_iterator noa =PftNIndA.find(type);
442  if (noa!=PftNIndA.end())
443  return 1.0/pow(noa->second,0.5);
444  }
445  if (layer==2){
446  map<string,int>::const_iterator nob =PftNIndB.find(type);
447  if (nob!=PftNIndB.end())
448  return 1.0/pow(nob->second,0.5);
449  }break;
450  };break;
451  // lower resource availability for intraspecific competition
452  case 2: if (layer==1)return NPftA/(1.0+NPftA);break;
453  // belowground
454  if (layer==2)return NPftB/(1.0+NPftB);break;
455  default: cerr<<"CCell::prop_res() - wrong input";exit(3);
456  }// end of switch
457  return -1; //should not be reached
458 }//end CCell::prop_res

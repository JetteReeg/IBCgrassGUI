154  * Vary the current individual's traits, based on a Gaussian distribution with a
155  * standard deviation of "ITVsd". Sub-traits that are tied will vary accordingly.
156  * Bounds on 1 and -1 ensure that no trait garners a negative value and keep the resulting
157  * distribution balanced. Other, trait-specific, requirements are checked as well. (e.g.,
158  * LMR cannot be greater than 1, memory cannot be less than 1).
159  */
160 void SPftTraits::varyTraits()

163  double dev;

167  //get the deviation from a gaussian distribution with a standard deviation of ITVsd
168  dev = CEnvir::normrand(0, SRunPara::RunPara.ITVsd);
169  // set the new value
170  LMR_ = LMR + (LMR * dev);
171  // make sure the value is ok
172  } while (dev < -1.0 || dev > 1.0 || LMR_ < 0 || LMR_ > 1);
173  LMR = LMR_;

653 {
654  int SumCells =SRunPara::RunPara.GetSumCells();
655  double CellScale =SRunPara::RunPara.CellScale();
656  // according to Schwinning and Parsons (1999): aboveground mass that is ungrazable
657  double ResidualMass =15300*SumCells*CellScale*CellScale*0.0001;
658  double MaxMassRemove, TotalAboveMass, MassRemoved=0;
659  double grazprob;
660 
661  double Max;
662 
663  TotalAboveMass=GetTotalAboveMass();
664 
665  //maximal removal of biomass
666  MaxMassRemove=TotalAboveMass*SRunPara::RunPara.PropRemove;
667  MaxMassRemove=min(TotalAboveMass-ResidualMass,MaxMassRemove);
668  // while the mass removed is lower than the maximal mass removed
669  while(MassRemoved<MaxMassRemove){
670  //calculate slope for individual grazing probability;
671  //sort PlantList descending after mshoot/LMR
672  sort(PlantList.begin(),PlantList.end(),CPlant::ComparePalat);
673  //plant with highest grazing susceptibility
674  CPlant* plant = *PlantList.begin() ;
675  Max = plant->mshoot*plant->Traits->GrazFac();
676 
677  random_shuffle(PlantList.begin(),PlantList.end());
678  // for all plants
679  plant_size i=0;
680  while((i<PlantList.size())&&(MassRemoved<MaxMassRemove)){
681  CPlant* lplant=PlantList[i];
682  // grazing probability is proportional to the highest grazing susceptibility
683  grazprob= (lplant->mshoot*lplant->Traits->GrazFac())/Max;
684  if (CEnvir::rand01()<grazprob) MassRemoved+=lplant->RemoveMass();
685  ++i;
686  }// end while there are plants
